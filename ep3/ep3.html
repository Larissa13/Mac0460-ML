def model(input1, input_size, label_size, layer1_size, layer2_size):
    W0 = tf.Variable(tf.truncated_normal([input_size,layer1_size], stddev=0.1))
    b0 = tf.Variable(tf.truncated_normal([layer1_size], stddev=0.1))
    W1 = tf.Variable(tf.truncated_normal([layer1_size, layer2_size], stddev=0.1))
    b1 = tf.Variable(tf.truncated_normal([layer2_size], stddev=0.1))
    W2 = tf.Variable(tf.truncated_normal([layer2_size, label_size], stddev=0.1))
    b2 = tf.Variable(tf.truncated_normal([label_size], stddev=0.1))

    layer1 = tf.add(tf.matmul(input1, W0), b0)
    layer1 = tf.nn.relu(layer1)
    layer2 = tf.add(tf.matmul(layer1, W1), b1)
    layer2 = tf.nn.relu(layer2)

    output = tf.matmul(layer2, W2) + b2
    return output

def perceptron(input_size, label_size, layer1_size, layer2_size, learning_rate, batch_size, training_size, display, random, data):

    input1 = tf.placeholder(tf.float32, shape=[None, input_size])
    label = tf.placeholder(tf.float32, shape=[None, label_size])

    apply_model = model(input1, input_size, label_size, layer1_size, layer2_size)
    cost = tf.reduce_mean(tf.nn.softmax_cross_entropy_with_logits(logits=apply_model, labels=label))
    optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate).minimize(cost)
    correct_prediction = tf.equal(tf.argmax(apply_model, 1), tf.argmax(label, 1))
    accuracy = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))

    n_batches = training_size//batch_size
    data_in = np.copy(data)
    if random == True:
        np.random.shuffle(data_in)
    labels_t = np.zeros((data_in.shape[0], 10))
    labels_t[np.arange(data_in.shape[0]), data_in[:, 0].flatten()] = 1
    in_err = []
    out_err = []
    init = tf.global_variables_initializer()

    with tf.Session() as sess:
        sess.run(init)

        for i in range(n_batches):
            _, c = sess.run([optimizer, cost], feed_dict={input1: data_in[i*batch_size:(i + 1)*batch_size, 1:],
                                                              label: labels_t[i*batch_size:(i + 1)*batch_size, :]})
            if (i*batch_size)%display == 0:
                correct_input = sess.run(accuracy, feed_dict={input1: data_in[:training_size, 1:],
                                                              label: labels_t[:training_size, :]})
                correct_val = sess.run(accuracy, feed_dict={input1: data_in[training_size:, 1:],
                                                              label: labels_t[training_size:, :]})

                in_err.append(1.0 - correct_input)
                out_err.append(1.0 - correct_val)
    return in_err, out_err

Comentários:¶

É fácil ver que o experimento que utiliza todos os pixels(1) tem erro menor que os outros dois. Provavelmente, porque há mais informação e de maneira que não seja muito prejudicada por overfitting.

Para estimar o delta foi usado que: P(A < b) = 1 - P(A >= b), assim é natural que P(|eout - ein| >= epsilon) = delta

Além disso, 10 mostrou-se uma quantidade pequena para estimar o delta, uma vez que ele por vezes estimou delta como sendo 1. Ou seja, uma estimação grosseira. Se o número de vezes fosse maior, poderíamos ter uma estimação mais precisa. Outra modificação que poderia ajudar seria estimar o epsilon com base em mais execuções dos testes também, ao invés de uma única.

Q2. O método de aprendizado adotado na questão anterior inclui regularização? Caso afirmativo, como? Caso negativo, como formularia a inclusão da regularização e porque esse procedimento melhoraria o resultado?

Resposta: O método utilizado não inclui regularização. Para incluir regularização poderíamos utilizar o método de decaimento de pesos ou "weight decay". Com isso é possível evitar que ruídos ou overfitting tornem o classificador enviesado e ineficaz. O conjunto de funções de hipótese (H) se torna mais restrito e menos volátil às variações nos dados de entrada. Para implementá-lo, utilizamos a heurística de Erro Aumentado. E_aug = E_in + penalty_factor onde penalty_factor = lambda * w'w Isto força o algoritmo de learning a escolher entre fazer o erro de entrada (e_in) muito pequeno e fazer os pesos muito pequenos. Como queremos minimizar o erro aumentado, isto força que pesos muito grandes sejam evitados.
